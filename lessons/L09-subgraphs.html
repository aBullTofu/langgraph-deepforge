class SubState(TypedDict): sub_input: str; sub_result: str
sub_builder = StateGraph(SubState); sub_builder.add_node(...)
subgraph = sub_builder.compile()

# 父图中包装调用
class ParentState(TypedDict): main_input: str; main_output: str
def call_sub(state: ParentState) -> dict:
    out = subgraph.invoke({"sub_input": state["main_input"]})
    return {"main_output": out["sub_result"]}
