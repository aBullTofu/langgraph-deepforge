⚔️ 实战编码

静态并行（同一 super-step 从多处出发）

builder.add_edge(START, "fetch_a"); builder.add_edge(START, "fetch_b"); builder.add_edge(START, "fetch_c")
# fetch_a, fetch_b, fetch_c 在同一 super-step 中自动并行执行！

Send API 动态并行（数量不定时）

from langgraph.types import Send
class State(TypedDict):
    queries: list[str]
    results: Annotated[list[str], operator.add]  # ⭐ Reducer 自动合并

def fan_out(state):  # 为每个 query 创建一个 Send
    return [Send("worker", {"query": q}) for q in state["queries"]]

def worker(state):  # 每个 worker 并行独立执行
    result = search(state["query"])
    return {"results": [result]}

builder.add_conditional_edges(START, fan_out, {"worker": "worker"})
