def act(state: AgentState) -> dict:
    """手动执行工具——可以加日志、权限检查、结果截断"""
    last_msg = state["messages"][-1]
    tool_msgs = []
    for tc in last_msg.tool_calls:
        print(f"[行动] {tc['name']}({tc['args']})")
        try:
            tool = tools_by_name.get(tc["name"])
            result = tool.invoke(tc["args"]) if tool else f"错误：未找到工具"
        except Exception as e:
            result = f"工具失败：{e}。请调整参数后重试。"  # ⭐ 错误返回给 LLM# Return error to LLM
        print(f"[观察] {result}")
        tool_msgs.append(ToolMessage(str(result), tool_call_id=tc["id"]))
    return {"messages": tool_msgs, "tool_calls": state.get("tool_calls", 0) + len(tool_msgs)}

防死循环

Preventing Infinite Loops

MAX_CALLS = 5
def should_continue(state) -> str:
    if state.get("tool_calls", 0) >= MAX_CALLS: return END
    last = state["messages"][-1]
    if hasattr(last, "tool_calls") and last.tool_calls: return "act"
    return END
